# Pre-Processing
df["Revenue"]=df["Revenue"].str.replace(',','.')

# Label Encoding
le = preprocessing.LabelEncoder()
df['Country'] = le.fit_transform(df['Country'])
df['NACE class'] = le.fit_transform(df['NACE class'].astype(str))
df['Revenue'] = le.fit_transform(df['Revenue'].astype('float64'))
df['Biggest Item'] = le.fit_transform(df['Biggest Item'].astype(str))
df['InputChannel'] = le.fit_transform(df['InputChannel'].astype(str))
df = df.dropna()

fpr3, tpr3, threshold = roc_curve(y_test, predictions3)
auc3 = auc(fpr3, tpr3)

plt.figure(figsize=(14, 7), dpi=100)

plt.plot(fpr1, tpr1, marker='^',color = "y", label='Naive Bayes Classifier (auc = %0.2f)' % auc1)
